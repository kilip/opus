# Opus Dash Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the PWA-capable frontend architecture for Opus Dash as described in ADR-003.

**Architecture:** A feature-based PWA using Vite, React 19, TanStack Router, TanStack Query, and Workbox for offline reads/writes. The project is split into `app/`, `routes/`, `features/`, and `shared/` directories.

**Tech Stack:** React 19, Vite, TanStack Router, TanStack Query, Tailwind CSS 4, vite-plugin-pwa, Vitest, React Testing Library.

// dash/src/shared/lib/api-client.ts
const BASE_URL = import.meta.env?.VITE_API_URL ?? 'http://localhost:8080';

export const apiClient = {
  get: async <T>(path: string): Promise<T> => {
    const res = await fetch(`${BASE_URL}${path}`);
    if (!res.ok) throw new Error(res.statusText);
    return res.json() as Promise<T>;
  },
  post: async <T>(path: string, body: unknown): Promise<T> => {
    const res = await fetch(`${BASE_URL}${path}`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(body),
    });
    if (!res.ok) throw new Error(res.statusText);
    return res.json() as Promise<T>;
  },
};
```

// dash/src/shared/hooks/useNetworkStatus.ts
import { useEffect, useState } from 'react';

export function useNetworkStatus() {
  const [isOnline, setIsOnline] = useState(navigator.onLine);

  useEffect(() => {
    const handleOnline = () => setIsOnline(true);
    const handleOffline = () => setIsOnline(false);
    window.addEventListener('online', handleOnline);
    window.addEventListener('offline', handleOffline);
    return () => {
      window.removeEventListener('online', handleOnline);
      window.removeEventListener('offline', handleOffline);
    };
  }, []);

  return { isOnline };
}
```

import { ReactNode, useState } from 'react';

export function Providers({ children }: { children: ReactNode }) {
  const [queryClient] = useState(() => new QueryClient({
    defaultOptions: {
      queries: {
        staleTime: 1000 * 60 * 5, // 5 minutes
      },
    },
  }));

  return (

  );
}
